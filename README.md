#School Management with TALL Stack_

##This is our first project with TALL stack.

#Learning By Doing.

#installation
    git clone https://github.com/Khantminaung313/School-Management.git

#composer update
    composer update

#compser install
    composer instal

#npm install
    npm install

#Copy .env file
    cp .env.example .env
    php artisan key:generate

#database migration
    php artisan migrate

#host your project
    npm run dev
    php artisan serve
